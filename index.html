Robust Speech Recognition Pattern

How to fix text duplication on mobile browsers

/**
 * ROBUST SPEECH-TO-TEXT STRATEGY
 * 
 * The Problem:
 * Standard `SpeechRecognition` with `continuous: true` is broken on
 * many Android devices. It causes the internal buffer to repeat
 * previous segments, resulting in "Text Text Text Text".
 * 
 * The Solution:
 * We use a "Manual Continuous Loop" strategy.
 * 
 * 1. CONFIGURATION
 *    Set `continuous = false`. This forces the engine to return
 *    a single final result and then stop.
 * 
 * 2. STATE MANAGEMENT
 *    Maintain two variables:
 *    - `fullTranscript`: A persistent string for the whole session.
 *    - `interimText`: Temporary text for words being spoken now.
 * 
 * 3. HANDLING RESULTS (onresult)
 *    - Loop through event results.
 *    - If `isFinal` is true: Append text to `fullTranscript`.
 *    - If `isFinal` is false: Set text to `interimText`.
 *    - UI Display = fullTranscript + " " + interimText.
 * 
 * 4. THE LOOP (onend)
 *    - When the engine stops (which it will do frequently because
 *      continuous is false), check your app's `isListening` flag.
 *    - If true, immediately call `recognition.start()`.
 * 
 * This bypasses the browser's internal buffer logic entirely,
 * giving you full control over the string concatenation.
 */

class SpeechHandler {
    constructor() {
        // Init with continuous FALSE to prevent buffer bugs
        this.recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)();
        this.recognition.continuous = false; 
        this.recognition.interimResults = true;
        
        this.fullTranscript = "";
        this.isListening = false;
        
        this.recognition.onresult = (e) => {
            let newInterim = "";
            for (let i = e.resultIndex; i < e.results.length; ++i) {
                if (e.results[i].isFinal) {
                    this.fullTranscript += " " + e.results[i][0].transcript;
                } else {
                    newInterim += e.results[i][0].transcript;
                }
            }
            console.log("Output:", this.fullTranscript + " " + newInterim);
        };

        this.recognition.onend = () => {
            if (this.isListening) this.recognition.start();
        };
    }

    start() {
        this.isListening = true;
        this.recognition.start();
    }
}
